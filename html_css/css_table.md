# 테이블에 색상을 주는 법
## 태그에 직접 하는법
    <td color:"blue"> 파란색 글씨 </td>
    <td background-color:"yellow">노란색 배경색</td>

## css
    <style>
      table{
          border-color:red;
          border-style:dashed;
       }
       td{
          border-color:blue;
          border-style:solid;
        }
    </style>


## 테이블 가운데 정렬하기(css)

    table{ 
        margin-left:auto;
        margin-right:auto;
     }
